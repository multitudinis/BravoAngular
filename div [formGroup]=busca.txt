  			this.http.get(`viacep.com.br/ws/${ CEP }/json/`).pipe(
    					catchError(erro => {
      			if (erro.status === 404) {
        		console.log('Produto nÃ£o localizado.');
      			}
    			  return throwError(() => new Error(erro));
  			  })
  			).subscribe(
  			  resultado => {
  			    console.log(resultado);
  			  }
  			);
  			 console.log(CEP)
  			}
